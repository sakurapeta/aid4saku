# 快捷键
alt + 鼠标滚轮 = 放大缩小
space + 鼠标滚轮 = 拖动
ctrl + 鼠标滚轮 = 水平移动
鼠标滚轮 = 垂直移动
ctrl + R = 标尺
ctrl + ; = 显示/隐藏标尺
ctrl + " = 网格
ctrl + Z = 撤回
ctrl + shift + Z = 重做
ctrl + Y = 轮廓
ctrl + alt + Y = 像素
ctrl + shift + Y = 叠印
tab = 隐藏工具栏
F = 隐藏工具栏和菜单栏
